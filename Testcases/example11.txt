int fibonacci(int n) 
{
    if (n <= 1) 
    {
        return n;
    } 
    else 
    {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

int main() {
    int n;
    int result;
    n = 10;
    result = fibonacci(n);
    printf("The %dth Fibonacci number is %d\n", n, result);
    return 0;
}
