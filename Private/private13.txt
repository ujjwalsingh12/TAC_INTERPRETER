int calculate_sum(int n) 
{
    int i;
    int sum;
    sum = 0;
    
    for (i = 0; i < n; i = i + 1) 
    {
        if (i < 5) 
        {
            sum = sum + i;
        } 
        else 
        {
            sum = sum + (i + 10);
        }
    }
    
    return sum;
}

int main() 
{
    int limit;
    int result;
    
    limit = 10;
    result = calculate_sum(limit);
    
    printf("Final Sum: %d\n", result);
    
    return 0;
}
