int sum_of_numbers(int n) 
{
    int i;
    int total;
    total = 0;
    
    for (i = 0; i < n; i = i + 1) 
    {
        total = total + i;
    }
    
    return total;
}

int main() 
{
    int result;
    
    result = sum_of_numbers(10);
    printf("Sum of numbers: %d\n", result);
    
    return 0;
}
